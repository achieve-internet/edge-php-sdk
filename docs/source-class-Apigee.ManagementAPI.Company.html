231:     /**
232:      * Saves this object's properties to the Edge server.
233:      *
234:      * If $force_update is set to true, we assume that this is an update call.
235:      * If it is false, we assume that it is an insert. If null is passed in,
236:      * we attempt an update, and if it fails we attempt an insert. This is
237:      * much less efficient, so declaring $force_update as a boolean will yield
238:      * faster response times.
239:      *
240:      * @param bool|null $force_update
241:      * @throws \Apigee\Exceptions\ResponseException
242:      * @throws \Exception
243:      */
244:     public function save($force_update = false)
245:     {
246:         // See if we need to brute-force this.
247:         if ($force_update === null) {
248:             try {
249:                 $this->save(true);
250:             } catch (ResponseException $e) {
251:                 if ($e->getCode() == 404) {
252:                     // Update failed because company doesn't exist.
253:                     // Try insert instead.
254:                     $this->save(false);
255:                 } else {
256:                     // Some other response error.
257:                     throw $e;
258:                 }
259:             }
260:             return;
261:         }
262:         $payload = array(
263:             'name' => $this->name,
264:             'displayName' => $this->displayName,
265:             'status' => $this->status,
266:             'attributes' => array()
267:         );
268:         if (count($this->attributes) > 0) {
269:             $payload['attributes'] = array();
270:             foreach ($this->attributes as $name => $value) {
271:                 $payload['attributes'][] = array('name' => $name, 'value' => $value);
272:             }
273:         }
274:         $url = null;
275:         if ($force_update || $this->createdAt) {
276:             $url = rawurlencode($this->name);
277:         }
278:         if ($force_update) {
279:             $this->put($url, $payload);
280:         } else {
281:             $this->post($url, $payload);
282:         }
283:         self::loadFromResponse($this, $this->responseObj);
284:     }

382:     private static function loadFromResponse(Company &$company, array $response)

384:         foreach ($response as $key => $value) {

388:                        $company->attributes[$name_value_pair['name']] = $name_value_pair['value'];

390:                 } else {

393:             }
