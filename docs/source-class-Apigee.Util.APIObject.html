139:   private function exec(\Guzzle\Http\Message\RequestInterface $request) {
140:     $this->responseCode = 0;
141:     $request->setAuth($this->config->user, $this->config->pass);
142:     try {
143:       $response = $request->send();
144:     }
145:     // catch ClientErrorResponseException, ServerErrorResponseException
146:     catch (\Guzzle\Http\Exception\BadResponseException $e) {
147:       $response = $e->getResponse();
148:     }
149:     catch (\Guzzle\Http\Exception\CurlException $e) {
150:       // Timeouts etc.
151:       DebugData::$raw = '';
152:       DebugData::$code = $e->getErrorNo();
153:       DebugData::$code_status = $e->getError();
154:       DebugData::$code_class = 0;
155:       DebugData::$exception = $e->getMessage();
156:       DebugData::$opts = array('request_headers' => $request->getRawHeaders());
157:       DebugData::$data = NULL;
158:       $exception = new ResponseException($e->getError(), $e->getErrorNo(), $request->getUrl(), DebugData::$opts);
159:       $exception->requestObj = $request;
160:       throw $exception;
161:     }
162:     $this->responseCode = $response->getStatusCode();
163:     $this->responseText = trim($response->getBody(TRUE));
164:     $this->responseLength = $response->getContentLength();
165:     $this->responseMimeType = $response->getContentType();
166:     $this->responseObj = array();
167:     $content_type = $response->getContentType();
168:     if (strpos($content_type, '/json') !== FALSE && (substr($this->responseText, 0, 1) == '{' || substr($this->responseText, 0, 1) == '[')) {
169:       $response_obj = @json_decode($this->responseText, TRUE);
170:       if (is_array($response_obj)) {
171:         $this->responseObj = $response_obj;
172:       }
173:     }
174: 
175:     $status = self::getStatusMessage($this->responseCode);
176:     $code_class = floor($this->responseCode / 100);
177: 
178:     DebugData::$raw = $this->responseText;
179:     DebugData::$opts = array(
180:       'request_headers' => $request->getRawHeaders(),
181:       'response_headers' => $response->getRawHeaders()
182:     );
183: 
184:     if ($request instanceof \Guzzle\Http\Message\EntityEnclosingRequestInterface) {
185:       DebugData::$opts['request_body'] = (string) $request->getBody();
186:     }
187:     DebugData::$opts['request_type'] = class_implements($request);
188:     DebugData::$data = $this->responseObj;
189:     DebugData::$code = $this->responseCode;
190:     DebugData::$code_status = $status;
191:     DebugData::$code_class = $code_class;
192:     DebugData::$exception = NULL;
193: 
194:     if ($code_class != 2) {
195:       $uri = $request->getUrl();
196:       if (!empty($this->responseCode) && isset($this->responseObj['message'])) {
197:         $message = 'Code: ' . $this->responseCode . '; Message: ' . $this->responseObj['message'];
198:       }
199:       else {
200:         $message = 'API returned HTTP code of ' . $this->responseCode . ' when fetching from ' . $uri;
201:       }
202:       DebugData::$exception = $message;
203:       self::$logger->error($this->responseText);
204: 
205:       // Create better status to show up in logs
206:       $status .= ': ' . $request->getMethod() . ' ' . $uri;
207:       if ($request instanceof \Guzzle\Http\Message\EntityEnclosingRequestInterface) {
208:         $body = $request->getBody();

212:         }
213:       }

216:       $exception->requestObj = $request;

218:       throw $exception;

330:   /**
331:    * Intercepts any snake_case method invocations that aren't already
332:    * defined, turns them into camelCase, and tries to invoke them.
333:    *
334:    * Incoming snake_case method names must contain no uppercase to
335:    * qualify for this transmogrification. In the interest of efficiency,
336:    * we also don't process any method names not containing an underscore.
337:    *
338:    * @TODO When we require PHP 5.4, make this a mix-in.
339:    *
340:    * @param string $method
341:    * @param array $args
342:    * @return mixed
343:    * @throws \Apigee\Exceptions\IllegalMethodException
344:    * @internal
345:    */
346:   public function __call($method, $args) {
347:     $class = get_class();
348: 
349:     if ($method == strtolower($method) && strpos($method, '_') !== FALSE) {
350:       $parts = explode('_', $method);
351:       $camel_case = array_shift($parts);
352:       foreach ($parts as $part) {
353:         $camel_case .= ucfirst($part);
354:       }
355:       if (method_exists($this, $camel_case)) {
356:         self::warnDeprecated($class, $method);
357:         return call_user_func_array(array($this, $camel_case), $args);
358:       }
359:       throw new IllegalMethodException('Class “' . $class . '” contains no such method “' . $method . '” (even after camelCasing)');
360:     }
361:     throw new IllegalMethodException('Class “' . $class . '” contains no such method “' . $method . '”');
362:   }

492:   /**
493:    * If payload is not already a string, convert it to a string based on its content-type.
494:    *
495:    * @static
496:    * @param $content_type
497:    * @param $payload
498:    */
499:   protected static function preparePayload($content_type, &$payload) {
500:     // If content_type includes charset, strip it off.
501:     if (($i = strpos($content_type, ';')) !== FALSE) {
502:       $content_type = trim(substr($content_type, 0, $i));
503:     }
504:     if ($content_type == 'application/json' && (is_object($payload) || is_array($payload))) {
505:       // Turn objects/arrays into JSON strings.
506:       $payload = json_encode($payload);
507:     }
508:     elseif ($content_type == 'application/xml') {
509:       // Turn XML document representations into strings.
510:       if ($payload instanceof \DOMDocument) {
511:         $payload = $payload->saveXML($payload->documentElement);
512:       }
513:       elseif ($payload instanceof \SimpleXMLElement) {
514:         $payload = $payload->asXML();
515:         // strip off processing instruction if present
516:         $payload = preg_replace('!^<\?[^?]+\?>!', '', $payload);
517:       }
518:     }
