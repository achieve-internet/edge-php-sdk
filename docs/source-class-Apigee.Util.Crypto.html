29:     /**
30:      * Sets the encryption key.
31:      *
32:      * Set the key before you call encrypt(). You must call decrypt() with the
33:      * same value as used for encrypt(). If the key length exceeds the maximum
34:      * (defined in the class constant MAX_KEY_LENGTH) it will be truncated.
35:      *
36:      * @static
37:      * @param string $key
38:      */
39:     public static function setKey($key)
40:     {
41:         if (strlen($key) > self::MAX_KEY_LENGTH) {
42:             $key = substr($key, 0, self::MAX_KEY_LENGTH);
43:         }
44:         self::$cryptoKey = $key;
45:         self::$encryptedStrings = array();
46:     }
47: 
48:     /**
49:      * Encrypts a string using the specified key.
50:      * Caches the result so that expensive encryption does
51:      * not need to happen more often than necessary.
52:      *
53:      * @static
54:      * @param string $string
55:      * @return string
56:      */
57:     public static function encrypt($string)
58:     {
59:         if (!isset(self::$encryptedStrings)) {
60:             self::$encryptedStrings = array();
61:         }
62:         if (!array_key_exists($string, self::$encryptedStrings)) {
63:             srand();
64:             $iv = mcrypt_create_iv(mcrypt_get_iv_size(MCRYPT_RIJNDAEL_128, MCRYPT_MODE_CBC), MCRYPT_RAND);
65:             $iv_base64 = rtrim(base64_encode($iv), '='); // Guaranteed to be 22 char long
66:             // Store password length so we can accurately trim in case of null-padding
67:             $encrypt = strlen($string) . "\n" . $string;
68:             $encrypted = mcrypt_encrypt(MCRYPT_RIJNDAEL_128, self::$cryptoKey, $encrypt, MCRYPT_MODE_CBC, $iv);
69:             self::$encryptedStrings[$string] = $iv_base64 . base64_encode($encrypted);
70:         }
71:         return self::$encryptedStrings[$string];
72:     }
73: 
74:     /**
75:      * Decrypts a string which was encrypted with Crypto::encrypt().
76:      *
77:      * @static
78:      * @param string $scrambled
79:      * @return string
80:      */
81:     public static function decrypt($scrambled)
82:     {
83:         $iv_base64 = substr($scrambled, 0, 22) . '==';
84:         $string_encrypted = substr($scrambled, 22);
85: 
86:         $iv = base64_decode($iv_base64);
87:         if ($iv === false || strlen($iv) < 16) {
88:             throw new \Apigee\Exceptions\ParameterException('Unable to parse encrypted string.');
89:         }
90:         $decrypted = mcrypt_decrypt(MCRYPT_RIJNDAEL_128, self::$cryptoKey, base64_decode($string_encrypted), MCRYPT_MODE_CBC, $iv);
91:         list ($length, $password) = explode("\n", $decrypted, 2);
92:         return substr($password, 0, intval($length));
93:     }
